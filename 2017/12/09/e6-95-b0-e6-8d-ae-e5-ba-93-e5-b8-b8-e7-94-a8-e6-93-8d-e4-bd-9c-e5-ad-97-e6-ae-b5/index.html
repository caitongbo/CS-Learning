数据库常用操作字段
数据库原理

发布日期:   2017-12-09
阅读次数:  

以家庭用户（用户姓名,个人信息,药物治疗,疾病史,过敏源）为例 //创建USERS表

CREATE TABLE USERS(USERS_NAME VARCHAR(10) NOT NULL,USERS_INFO VARCHAR(25) NOT NULL,USERS_MEIDE VARCHAR(25),USERS_DISEASE VARCHAR(25),USERS_ALLERGY VARCHAR(25),PRIMARY KEY(USERS_NAME,USERS_INFO));

//插入一条姓名为李凯，用户信息为310000199905263540，药物名称为999感冒灵颗粒，疾病名称为感冒，过敏源为花粉的用户记录

INSERT INTO USERS VALUES(‘李凯’,’310000199905263540’,’999感冒灵颗粒’,’感冒’,’花粉’);

//将用户信息为3100000199905263540的疾病名称修改为发烧

UPDATE USERS SET USERS_DISEASE=’发烧’ WHERE USERS_INFO=’310000199905263540’;

//删除姓名为李凯的记录

DELETE FROM USERS WHERE USERS_NAME=’李凯’;

//增加一个名为USERS_AGE，值为男的字段

ALTER TABLE USERS ADD (AGE VARCHAR(10) DEFAULT ‘男’ );

//修改一个名为USERS_AGE字段

ALTER TABLE USERS MODIFY(USERS_AGE VARCHAR(25) DEFAULT ‘UNKNOWN’);

//删除一个名为USERS_AGE的字段

ALTER TABLE USERS DROP(USERS_AGE);

//删除USERS表

DROP TABLE USERS;

//查询用户列表

SELECT * FROM USERS;

//查看姓名列表并按升序排列

SELECT USERS_NAME FROM USERS ORDER BY USERS_NAME ASC;

//创建视图USERS_1查看姓名列表并按升序排列

CREATE OR REPLACE VIEW USERS_!
AS
SELECT USERS_NAME FROM USERS ORDER BY USERS_NAME ASC
WITH READ ONLY;
SELECT * FROM USERS_1;
