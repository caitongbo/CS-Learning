1. 编写一个圆类 Circle, 其中有圆周率、 半径，其中圆周率为 static 成员。此外,Circle 中还有两个方法分别是计算圆的面积和周长。编写一个测试程序，并创建一个 Circle 类型的对象, 并输出该圆的面积和周长。

public class CacCir {
public static void main(String args[]){
Circle a=new Circle(5);
System.out.println(a.CaculateSc());
System.out.println(a.CaculateCc());
}
}
class Circle{
double t=3.1415926;
double r=0.0;
Circle(int x) {
r=x;
}
public double CaculateSc(){
double Sc=r*r*t;
return Sc;
}
public double CaculateCc(){
double Cc=2*t*r;
return Cc;
}
}
