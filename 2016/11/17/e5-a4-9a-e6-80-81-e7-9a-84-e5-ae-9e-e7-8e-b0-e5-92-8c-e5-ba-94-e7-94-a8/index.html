1. 建立类Datasort,该类包括两个相同的方法sort,分别实现对两个整数和三个整数进行大小排序并输出排序的结果。建立测试类Test建立实例测试程序。

public class Test {
public static void main(String args[]){
Datasort T=new Datasort();
T.sort(1,6);
T.sort(10,14,89);
}
}
class Datasort{
public void sort(int a,int b){
if(a>b)
System.out.println(“两个数由大到小顺序是：”+a+” “+b);
else System.out.println(“两个数由大到小顺序是：”+b+” “+a);
}

public void sort(int c,int d,int e){
 int t=c;
    if(c<d){
        t=c;
        c=d;
        d=t;
    }
    if(c<e){
        t=e;
        e=c;
        c=t;
    }
    if(d<e){
        t=d;
        d=e;
        e=t;
    }
    System.out.println("三个数由大到小顺序是："+c+" "+d+" "+e);
}

}
