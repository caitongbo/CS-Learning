#include<stdio.h>

#include<stdlib.h>

#define MAXSIZE 100

#define TRUE 1

#define FALSE 0
typedef struct
{
int row,col;
int e;
}Triple;
typedef struct
{
Triple data[MAXSIZE+1];
int m,n,len;
}TSMatrix;

TSMatrix CreateMatrix()//构造稀疏矩阵
{
TSMatrix \M=(TSMatrix *)malloc(sizeof(TSMatrix));
int m,n,num;
int i;
printf(“请输入您想要的稀疏矩阵的行，列，非零元素个数。\n”);
scanf(“%d,%d,%d”,&m,&n,&num);

// M->data=(Triple )malloc(sizeof(Triple)num);
M->m=m;
M->n=n;
M->len=num;
i=0;
printf(“开始矩阵构造！\n”);
while(i<num)
{
scanf(“%d %d %d”,&M->data[i].row,&M->data[i].col,&M->data[i].e);
if(M->data[i].row>m||M->data[i].col>n)
{
printf(“构造稀疏矩阵失败!\n”);
exit(1);
}
i++;
}
return M;
}

{
int p;

void Change(TSMatrix N)
{
int i,j;
int k;

{
int q=0;
int col,p;
